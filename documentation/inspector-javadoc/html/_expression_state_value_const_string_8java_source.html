61  public static String convertToString (String str) throws JPFInspectorParsingErrorException {
62  assert (str != null);
63 
64  StringBuilder sb = new StringBuilder(str.length());
65 
66  CharParsingState sps = new CharParsingState(str, '"');
67 
68  // Check initial "
69  if (sps.strArr.length < 1) {
70  throw new JPFInspectorParsingErrorException("Empty string", str, 0);
71  }
72 
73  if (sps.strArr[0] != '\"') {
74  throw new JPFInspectorParsingErrorException("Error while parsing character at 0. The double quote (\") character is expected in the beginning.", str, 0);
75  }
76  sps.pos++;
77 
78  int bound = sps.strArr.length - 1; // The last character should be \"
79  while (sps.pos < bound) {
80  char nextChar = ExpressionStateValueConstChar.parseCharRepresentation(sps);
81  sb.append(nextChar);
82  }
83 
84  if (sps.pos >= sps.strArr.length) {
85  throw new JPFInspectorParsingErrorException("Error while parsing string at " + (str.length())
86  + ". The terminating double quote (\") character is expected.", str, str.length());
87  }
88 
89  if (sps.strArr[sps.pos] != '\"') {
90  throw new JPFInspectorParsingErrorException("Error while parsing string at " + (sps.pos + 1)
91  + ". The terminating double quote (\") character is expected as a last character in string.", str, sps.pos + 1);
92  }
93 
94  if (sps.pos < sps.strArr.length - 1) {
95  throw new JPFInspectorParsingErrorException("Error while parsing string at " + (sps.pos + 1) + ". Garbage after terminating double quote (\").", str,
96  sps.pos + 1);
97  }
98 
99  return sb.toString();
100  }
