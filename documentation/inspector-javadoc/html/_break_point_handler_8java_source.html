145  private BreakPointStatus createBreakPointImpl(BreakPointCreationInformation newBP,
146  ExpressionBooleanInterface newBPExpression,
147  boolean hidden,
148  boolean firstHit) throws JPFInspectorGenericErrorException {
149  assert newBP != null;
150  assert newBPExpression != null;
151 
152  synchronized (breakpoints) {
153  InternalBreakpointHolder ibp = null;
154  if (newBP.getBPID() != BreakPointCreationInformation.BP_ID_NOT_DEFINED) {
155  ibp = breakpoints.get(newBP.getBPID());
156  if (ibp == null) {
157  throw new JPFInspectorGenericErrorException("Breakpoint with ID=" + newBP.getBPID() + " not exists --> cann't be modified");
158  }
159  }
160  if (ibp == null) {
161  ibp = new InternalBreakpointHolder(newBP.getBPID(), callBacks, !hidden, firstHit);
162  }
163  ibp.modifyBPSettings(newBP, newBPExpression);
164  breakpoints.put(ibp.getBPID(), ibp);
165  // Check for error
166  if (newBP.getBPExpression() != null) {
167  if (!newBP.getBPExpression().trim().isEmpty()) {
168  return null;
169  }
170  }
171  return ibp.getBreakpointStatus(null);
172  } // end sync
173  }

175  private AssertStatus createAssertImpl (AssertCreate newAssert, ExpressionBreakpointAssert newBPExpression) throws JPFInspectorGenericErrorException {
176  assert newAssert != null;
177  assert newBPExpression != null;
178 
179  synchronized (breakpoints) {
180  InternalAssertHolder iah = null;
181  if (newAssert.getBPID() != BreakPointCreationInformation.BP_ID_NOT_DEFINED) {
182  InternalBreakpointHolder ibp = breakpoints.get(newAssert.getBPID());
183  if (ibp == null) {
184  throw new JPFInspectorGenericErrorException("Assertion with ID=" + newAssert.getBPID() + " not exists --> cann't be modified");
185  }
186  if (ibp.isAssert() == false) {
187  throw new JPFInspectorGenericErrorException("ID=" + newAssert.getBPID() + " does not represent assert");
188  }
189 
190  assert (ibp instanceof InternalAssertHolder);
191  iah = (InternalAssertHolder) ibp;
192  }
193  if (iah == null) {
194  iah = new InternalAssertHolder(newAssert.getBPID(), callBacks, true, false, newAssert.getPosition(), newAssert.getCondition());
195  }
196 
197  iah.modifyBPSettings(newAssert, newBPExpression);
198 
199  breakpoints.put(iah.getBPID(), iah);
200 
201  // Check for error
202  if (newAssert.getBPExpression() != null) {
203  if (!newAssert.getBPExpression().trim().isEmpty()) {
204  return null;
205  }
206  }
207  return iah.getBreakpointStatus(null);
208  } // end sync

243  public boolean checkBreakpoints (InspectorState inspState) {
