36  public ExpressionBreakpointThreadScheduled (BreakPointModes bpMode, Integer threadNum) {

46  @Override
47  public boolean evaluateExpression (InspectorState state) {
48  // Notice We handle "threadScheduled"
49  // in this case the getLastThread gets Newly scheduled thread and
50  // lastTransition (if exists), holds previous thread
51  assert state != null;
52  if (state.getListenerMethod() != ListenerMethod.LM_THREAD_SCHEDULED) {
53  return false;
54  }
55  if (threadNum == null) {
56  return true;
57  }
58 
59  VM vm = state.getJVM();
60  if (bpMode == BreakPointModes.BP_MODE_THREAD_SCHEDULED_BOTH) {
61  Transition tr = vm.getLastTransition();
62  ThreadInfo prevTi = (tr != null ? tr.getThreadInfo() : null);
63 
64  return ((MigrationUtilities.getLastThreadInfo(vm).getId() == threadNum) // Newly planned thread
65  || ((prevTi != null) && (prevTi.getId() == threadNum))); // / Previous thread if exists
66 
67  } else if (bpMode == BreakPointModes.BP_MODE_THREAD_SCHEDULED_IN) {
68  return MigrationUtilities.getLastThreadInfo(vm).getId() == threadNum;
69 
70  } else if (bpMode == BreakPointModes.BP_MODE_THREAD_SCHEDULED_OUT) {
71  Transition tr = vm.getLastTransition();
72  if (tr == null) {
73  return false;
74  }
75  ThreadInfo prevTi = tr.getThreadInfo();
76  return prevTi.getId() == threadNum;
77  }
78 
79  return false;
80  }
