62  assert state != null;

64  if (state.getListenerMethod() != ListenerMethod.LM_INSTRUCTION_EXECUTED) {
65  return false;
66  }

68  VM vm = state.getJVM();

70 
71  if (!(inst instanceof FieldInstruction)) {

74 
75  FieldInstruction fiInst = (FieldInstruction) inst;
76  if (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS) {
77  return fn.isSameField(fiInst.getFieldInfo());
78  } else if (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS_READ) {
79  return (fiInst instanceof GETFIELD) || (fiInst instanceof GETSTATIC) && fn.isSameField(fiInst.getFieldInfo());
80  } else if (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS_WRITE) {
81  return (fiInst instanceof PUTFIELD) || (fiInst instanceof PUTSTATIC) && fn.isSameField(fiInst.getFieldInfo());
82  }
83 
84  return false;
85  }
