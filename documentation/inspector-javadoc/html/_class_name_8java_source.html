77  /*
78  * Converts name of the class of file into regular expression patter.
79  * <br>
80  * The name may contain * to mark (eventually empty) number of arbitrary characters.
81  * Note that "." (and other special regexp characters) are escaped
82  * so that the * has special meaning.
83  * <br>
84  * Examples:
85  * <br> "MyEX*.java" is converted into MyEx.*\.java.
86  *
87  * @param name Name to convert
88  * @return regular expresison pattern
89  */
90  static public String convertName2RegExp(String name) {
91 
92  // Conversions ... we permit only * in regexp
93  // '\' -> "\\"
94  // '[' -> "\["
95  // ']' -> "\]"
96  // '}' -> "\{"
97  // '{' -> "\}"
98  // '(' -> "\("
99  // ')' -> "\)"
100  // '+' -> "\+"
101  // '^' -> "\^"
102  // '$' -> "\$"
103  // '?' -> "\?"
104  // ',' -> "\,"
105  // '&' -> "\&"
106  // '|' -> "\|"
107  name = name.replace( "\\", "\\\\");
108  name = name.replace( "[", "\\[");
109  name = name.replace( "]", "\\]");
110  name = name.replace( "}", "\\}");
111  name = name.replace( "{", "\\{");
112  name = name.replace( "(", "\\(");
113  name = name.replace( ")", "\\)");
114  name = name.replace( "+", "\\+");
115  name = name.replace( "^", "\\^");
116  name = name.replace( "$", "\\$");
117  name = name.replace( "?", "\\?");
118  name = name.replace( ",", "\\,");
119  name = name.replace( "&", "\\&");
120  name = name.replace( "|", "\\|");
121 
122  // '.' -> "\." not any char
123  // '*' -> ".*"
124  name = name.replace( ".", "\\.");
125  name = name.replace( "*", ".*");
126  return "^" + name + "\\z";
127  }
