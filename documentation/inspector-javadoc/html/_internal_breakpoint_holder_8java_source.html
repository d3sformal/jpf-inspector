163  public boolean evaluateBreakpoint (InspectorState state) {
164  boolean bpHitted = false;
165 
166  // Hit condition
167  if (bpExpression != null) {
168  try {
169  bpHitted = bpExpression.evaluateExpression(state);
170  } catch (JPFInspectorException e) {
171  // Ignore the exception Breakpoint cannot be evaluated --> thus does not hold
172  }
173  }
174 
175  if (bpHitted) {
176  bpHitCounter++;
177  bpHitCounterTotal++;
178  }
179 
180  boolean bpShouldExecuteAction = bpHitted;
181  if (lowerBound != null && bpHitCounter < lowerBound) {
182  bpShouldExecuteAction = false;
183  }
184 
185  if (upperBound != null && bpHitCounter > upperBound) {
186  bpShouldExecuteAction = false;
187  }
188 
189 
190  MethodInfo topFrameMethodInfo = state.getJVM().getCurrentThread().getTopFrameMethodInfo();
191  if (bpShouldExecuteAction && topFrameMethodInfo != null) {
192  if (log.isLoggable(Level.FINE)) {
193  log.fine("Now checking within class: " + topFrameMethodInfo.getClassName());
194  }
195  if (InspectorConfiguration.getInstance().isClassIgnored(topFrameMethodInfo.getClassName())) {
196  bpShouldExecuteAction = false;
197  }
198  }
199  if (bpShouldExecuteAction && isUserBreakpoint()) {
200  if (bpState == BreakPointStates.BP_STATE_DISABLED) {
201  // No action expected
202  } else if (bpState == BreakPointStates.BP_STATE_LOGGING) {
203  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
204  } else if (bpState == BreakPointStates.BP_STATE_ENABLED) {
205  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
206  } else {
207  throw new RuntimeException("Unknown " + BreakPointStates.class.getSimpleName() + " entry " + bpState);
208  }
209  }
210 
211  return bpShouldExecuteAction && bpState == BreakPointStates.BP_STATE_ENABLED;
212 
213  }
