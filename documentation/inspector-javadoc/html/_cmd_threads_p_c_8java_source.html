45  assert inspector != null;

47  Map<Integer, InstructionPosition> places = inspector.getThreadsPC(threadNum);
48 
49  if (places == null) {

55  return;
56  }
57 
58  for (Entry<Integer, InstructionPosition> entry : places.entrySet()) {

60  }

64  }

89  private static String instructionPosition2String(InstructionPosition ip) {
90  StringBuilder sb = new StringBuilder(200);
91  sb.append(ip.getFileName());
92  sb.append(':');
93  sb.append(ip.getLineNumber());
94  String sourceLine = ip.getSourceLine();
95  if (sourceLine != null && !sourceLine.trim().isEmpty()) {
96  sb.append(':');
97  sb.append(sourceLine);
98  }
99  String className = ip.getClassName();
100  if (className != null && !className.trim().isEmpty()) {
101  sb.append("\n\t");
102  sb.append(className);
103  sb.append(':');
104  sb.append(ip.getMethodName());
105  sb.append(':');
106  sb.append(ip.getInstructionOffset());

109  }
