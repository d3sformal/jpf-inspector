61  public String expressError (int maxLineLength) {

75  for (int i = 0; i < linesToPrint - 1; i++) {
76  sb.append(expression.substring(i * maxLineLength, (i + 1) * maxLineLength));
77  sb.append('\n');
78  }
79

81  // command text...

85  sb.append("...\n");

92  sb.append(' ');

101  // the last line -> command text
102  // marker line ---> texti... ^
103

105  int lastLineMaxLen = maxLineLength - LAST_LINE_SPACE - 3;
106  if (lastLineMaxLen < 0) {
