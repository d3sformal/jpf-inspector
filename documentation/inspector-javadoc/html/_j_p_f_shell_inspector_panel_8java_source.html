145  s = INITIAL_COMMANDS[initialCmdProcessed++];
146  } else {
147  s = console.readLine(Constants.PROMPT);
148  }
149  String sTrimmed = s.trim();
150  if (sTrimmed.length() == 0) {
151  consolePrintStream.println("\n" + Constants.PROMPT);
152  }
153  if (s.trim().length() > 0) {
154  consolePrintStream.println();
155  inspector.executeCommand(s);
156  }
157  } catch (IOException e) {

159  }
160 
161  } // end of while
162  } catch (InterruptedException e) {
163  // Terminate the command loop
164  }
